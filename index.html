// ══ LocalStorage helpers ══

function starKey(langId, catId, level) { return `lf_star_${langId}_${catId}_${level}`; }

function getStarred(langId, catId, level) {

  try { return JSON.parse(localStorage.getItem(starKey(langId,catId,level)) || '[]'); }

  catch { return []; }

}

function saveStarred(langId, catId, level, ids) {

  try { localStorage.setItem(starKey(langId,catId,level), JSON.stringify(ids)); } catch {}

}

function isStarred(id) {

  const { lang, cat, level } = state;

  return getStarred(lang.id, cat.id, level).includes(id);

}

function toggleStarId(id) {

  const { lang, cat, level } = state;

  let ids = getStarred(lang.id, cat.id, level);

  if (ids.includes(id)) ids = ids.filter(x => x !== id);

  else ids.push(id);

  saveStarred(lang.id, cat.id, level, ids);

}

async function selectLevel(level, pillEl) {

  document.querySelectorAll('.level-pill').forEach(p => p.classList.remove('active'));

  pillEl.classList.add('active');

  state.level = level;

  state.starFilter = false;

  state.order = 'seq';

  try {

// ══ Build display list ══

function buildDisplayCards() {

  let cards = [...state.allCards];

  if (state.starFilter) {

    const ids = getStarred(state.lang.id, state.cat.id, state.level);

    cards = cards.filter(c => ids.includes(c.id));

  }

  if (state.order === 'rnd') {

    for (let i = cards.length-1; i > 0; i--) {

      const j = Math.floor(Math.random()*(i+1));

      [cards[i],cards[j]] = [cards[j],cards[i]];

    }

  }

  state.displayCards = cards;

  state.idx = 0;

}
